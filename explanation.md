navigate to yolo create docker-composer.yaml
 add the following 

 version: '3.8'

services:
  yolo.client:
    build: ./client
    ports:
      - "3000:3000"
    networks:
      -  yolo_network

  backend:
    build: ./backend
    ports:
      - "5000:5000"
    networks:
      - yolo_network

networks:
  yolo_network:
    driver: bridge
